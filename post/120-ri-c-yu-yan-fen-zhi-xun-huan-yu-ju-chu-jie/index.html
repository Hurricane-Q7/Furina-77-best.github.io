循环语句

#include<stdio.h
int main()
{
    int ch = getchar();//getchar的返回失败的时候返回值就是是EOF(end of file),其实就是-1，-1是个整形数据，最好用int,当然也可以用char!getchar返回成功是返回值就是字符的ASCII值！
    printf("%c\n",ch);
    putchar(ch);
    return 0;
}

int main()
{
    int ch = 0;
    while((ch=getchar())!=EOF)//getchar实际上接收了两个字符，'a'与'\n'
    //显然两个的值都不等于-1,putchar会输出两次，一次是'a'，另一次是'\n',实现了自动换行。
    {
        putchar(ch);
    }
    return 0;

}

getchar与scanf

C语言中，scanf 和 getchar 这类输入函数，不会直接从键盘读取数据，而是从「输入缓冲区」中读取数据。我们从键盘上输入东西时，会先放到输入缓冲区里面，然后scanf与getchar会主动去读取!

  • scanf("%s") 的规则：只读取「非空白字符」，遇到 空格 / 制表符 / 回车（\n）这类空白字符就会停止读取。
  • getchar来者不拒，照单全收，但是只能读取一个字符!
    关于缓冲区的代码

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
int main()
{
	char password[20] = { 0 };
	printf("请输入密码:>");
	scanf("%s", password);//输入123456'\n',先放在缓冲区里面，根据scanf的规则，scanf读取123456,getchar读取回车符!
	printf("请确认密码(Y/N):>");
	int ret = getchar();
	if ('Y' == ret)//这里显然不相等，执行else部分
	{
		printf("YES\n");
	}
	else {
		printf("No\n");//输出NO,可见这个代码并没有实现让确认密码的功能!
	}
	return 0;//缓冲区不干净!
}

#define _CRT_SECURE_NO_WARNINGS

int main()
