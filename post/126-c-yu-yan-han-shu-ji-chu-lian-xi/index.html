第一题输出100~200之间的素数

Definition :素数是质数的别称，大于1的自然数中，除了1和它自身外，不能被其他自然是整除的数称为素数！

第一版:
int main()
{
	//flag==1表示是素数，等于表示不是素数。
	int count = 0;
	int j = 2;
	int i = 100;
	printf("100~200内的素数有：\n");
	for (i = 100; i <= 200; i++)
	{
		int flag = 1;//对产生的每一个数字先假设它是素数
		for (j = 2; j < i; j++)
		{
			if (i % j == 0)
			{
				flag = 0;
				break;
			}
	   }
		if (flag == 1)
		{
			count++;
			printf("%d ", i);
		}
	}
	printf("\n100~200内的素数一共有%d个!", count);
	return 0;
}

最终优化版:
int main()
{
	//flag==1表示是素数，等于表示不是素数。
	int count = 0;
	int j = 2;
	int i = 100;
	printf("100~200内的素数有：\n");
	for (i = 101; i <= 200; i+=2)//i+=2表示跳过偶数，直接处理奇数，因为偶数不可能是素数
	{
		int flag = 1;//对产生的每一个数字先假设它是素数
		for (j = 2; j < i; j++)//也可以写，j<sqrt(i)
		{
			if (i % j == 0)
			{
				flag = 0;
				break;
			}
	   }
		if (flag == 1)
		{
			count++;
			printf("%d ", i);
		}
	}
	printf("\n100~200内的素数一共有%d个!", count);
	return 0;
}

写一个函数判断一年是不是闰年

什么是闰年?

闰年是为了弥补公历纪年中因人为历法规定造成的年度天数与地球实际公转周期的时间差而设立的
生成1000~2000年之间的闰年
判断闰年的规则：

  1. 能被4整除但是不能被100整除。
  2. 能被400整除是闰年。
int main()
{
    int year = 0;
    for(year = 1000;year<=2000;year++)
    {
        if(year %4 == 0)
    {
        if(year % 100 != 0)
        {
           printf("%d ",year);
        }
    }
    }
    if(year%400 == 0)
    {
        printf("%d ",year);
    }
}

int is_leap_year(int y)
{
	if (((y % 4 == 0) && (y % 100 != 0)) || (y % 400 == 0))
	{
		return 1;
	}
	else
	{
		return 0;
	}
}//设计函数要确保它的功能的单一、独立性，力求实现一个功能即可，高内聚，低耦合！
//高内聚就是让一个模块的所有代码都围绕一个单一、明确的核心职责展开，不掺杂无关的功能，做到 “专事专办”。
//低耦合就是让不同模块之间的交互尽可能少、简单、间接，模块之间尽量 “互不干涉”，一个模块的修改，不会连锁影响其他模块（即减少 “牵一发而动全身”）。

写一个函数实现整形有序数组的二分查找

#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
//二分查找，折半查找法
int binary_search(int arr[],int target,int sz)
{
	int left_idx = 0;
	int right_idx = sz - 1;
	while (left_idx <= right_idx)//left_idx<=right_idx说明还有查找空间
	{
		int mid = (left_idx + right_idx) / 2;
		if (arr[mid] < target)//说明目标元素在arr[mid]的右边，
		{
			left_idx = mid + 1;//变动左下标，范围向右边缩小
		}
		else if (arr[mid] > target)//说明目标元素在arr[mid]的左边
		{
			right_idx = mid - 1;//变动右下标，范围向左边缩小
		}
		else//arr[mid]==target
		{
			return mid;//返回下标mid
		}
		
	}
	if (left_idx > right_idx)
		{
			return - 1;
		}
}
int main()  
{
	int target = 0;//要查找的数字
	int arr[10] = { 0 };
	int i = 0;
	printf("输入10个数字，按升序排列:");
	//输入
	for (i = 0; i < 10; i++)
	{
		scanf("%d", &arr[i]);
	}
	int sz = sizeof(arr) / sizeof(arr[0]);//计算元素个数
	printf("请输入你要查找的数字：");
	scanf("%d", &target);
	int ret = binary_search(arr, target, sz);//自定义二分查找函数，专门实现查找功能，利用返回值判断查找情况;
	if (ret == -1)
		//数组下标不可能为负数
	{
		printf("找不到!\n");
	}
	else
	{
		printf("找到了，下标是%d\n", ret);//打印下标
	}
	return 0;
}

写一个函数实现，调用几次这个函数，就输出调用的次数！

void add(int* pn)//pn=&num;*pn==num
{
	printf("hello world!\n");
	(*pn)++;
}
int main()
{
	int num = 0;
	int i = 0;
	for (i = 0; i < 10; i++)
	{
		add(&num);
	}
	printf("一共调用了%d次add函数!\n", num);
	return 0;
}
