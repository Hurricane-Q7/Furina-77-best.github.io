第四题

演示多个字符从两端移动，向中间汇聚

#include <stdio.h>
#include<string.h>
#include<windows.h>
#include<stdlib.h>
int main()
{
	char arr1[]="welcome to bit lesson!";
	char arr2[]="######################";
	int left = 0;//计算最小下标与最大下标
	int right = strlen(arr1)-1;//自觉排除'\0'。
	while(left<=right)
	{
		arr2[left]=arr1[left];
		arr2[right]=arr1[right];
		printf("%s\n",arr2);
		Sleep(1000);//让程序慢点，1s=1000ms.
		system("cls");//system是<stdlib.h>中的一个库函数，用来调用OS的命令行。
		left++;
		right--;
	}
	printf("%s", arr2);
	return 0;
}

第五题

模拟用户登入（只允许登入三次密码）

int main()
{
	int i = 0;
	char password[20] = { 0 };
	
	for (i = 0; i < 3; i++)
	{
		printf("请输入密码:->");
		scanf("%s", password);
		if (strcmp(password, "abcde") == 0)
		{
			printf("密码正确!\n");
			break;
		}
		printf("密码错误!\n");
	}
	if (i == 3)
	{
		printf("三次密码均错误，退出程序!");
	}
	return 0;
}
