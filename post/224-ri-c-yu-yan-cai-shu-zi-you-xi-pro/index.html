int is_valid(int guess[])//检验玩家输入是否合法（首位非零，没有重复数字）,返回1说明合法，返回0说明不合法
{
	int flag = 0;
	int num = 0;
	int i = 0;
	int used[4] = { 0 };
	while (i == 0)
	{
		if (guess[i] != 0)
		{
			num = guess[i++];
			used[num] = 1;
		}
		else
		{
			return 0;
		}
	}
	for (i = 0; i < 4; i++)
	{
		if (guess[i] == guess[i + 1])
		{
			flag = flag + 1;
		}
	}
	if (flag >= 1)
	{
		return 0;
	}
}
