switch语句

switch语句也是一种分支语句，常常用于多分支的情况!

switch(整形表达式,整形数值)
{
    case 整形常量表达式: 语句；（可以是字符，字符以ASCII值存储）break;
    eg: case 1:printf("Monday");break;//break;就是中止，截断的意思，没有break的话，程序会从当前case项，向下穿透，执行后面的case项对应的语句直到遇到break
    default子句，用来处理上面多种case都没有包含的情况，一般为异常情况。
}

一个简单的例子：
#include <stdio.h>
int main()
{
    int day = 0;
    scanf("%d",&day);
    switch(day)
    {
        case 1:printf("1");break;
        case 2:printf("2");break;
        case 3:printf("3");break;
        default :printf("error!");break;
    }
    return 0;
}

#include<stdio.h>
int main()
{
	int n = 1;
	int m = 2;
	switch(n)
	{
		case 1:m++;
		case 2:n++;
		case 3:
			switch (n) 
			{
				case 1:n++;
				case 2:m++;n++;
				break;		
			}
		case 4:m++;break;
		default:break;
	}
	printf("m = %d,n = %d\n",m,n);
	return 0;
}

int func(int a)
{
	int b;
	switch (a)
	{
	case 1:b = 30;
	case 2:b = 20;
	case 3:b = 16;//无break，也会穿透到default
	default:b = 0;
	}
	return b;
}
int main()
{
	int ret = func(1);
	printf("%d", ret);
	return 0;
}

总结：switch语句是可嵌套的，break语句只能跳出当前switch语句的范围

循环语句

循环语句中的break与continue!
break是彻底停止整个循环（永久性）。
continue是跳过本次循环，直接执行下一次循环，continue一旦执行，跳过continue后面的代码，直接去判断部分，进行下一次循环的判断。

#include<stdio.h> 
int main()
{
    int i=1;
    while (i<=10)
    {
        if(5==i)
        {
            continue;
        }
        printf("%d ",i);
        i=i+1;
    }
    return 0;
}//输出结果:
